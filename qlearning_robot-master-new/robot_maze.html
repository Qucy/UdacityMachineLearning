import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    if (random.random() < epsilon):
        return actions[random.randint(0, len(actions)-1)] # random choose action
    else:
        return max(qline, key=lambda i: qline[i])  # greed choose
    
# Test choose_action
total_step = 100000
num_of_left = 0
num_of_other_action = 0
for i in range(total_step):
    if (choose_action(epsilon) == 'l'):
        num_of_left += 1
    else:
        num_of_other_action += 1
print("Probability go left:{}, Probability go other action:{}".format(num_of_left/total_step, num_of_other_action/total_step))
